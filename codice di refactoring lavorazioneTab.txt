// ===== FILE: buste/[id]/_components/tabs/LavorazioneTab.tsx =====
'use client';

import { useState, useEffect } from 'react';
import { createBrowserClient } from '@supabase/ssr';
import { Database } from '@/types/database.types';
import { mutate } from 'swr';
import {
  Factory,
  Plus,
  Clock,
  X,
  Truck,
  ClipboardCheck,
  Wrench,
  Stethoscope,
  Sparkles,
  Trash2,
  Loader2,
  CheckSquare,
  Calendar,
  AlertCircle,
  ArrowRight
} from 'lucide-react';
import { useUser } from '@/context/UserContext';

// --- TYPES ---
type BustaDettagliata = Database['public']['Tables']['buste']['Row'] & {
  clienti: Database['public']['Tables']['clienti']['Row'] | null;
  profiles: Pick<Database['public']['Tables']['profiles']['Row'], 'full_name'> | null;
  controllo_profile: Pick<Database['public']['Tables']['profiles']['Row'], 'full_name'> | null;
  status_history: Array<any>;
};

type Lavorazione = {
  id: string;
  busta_id: string;
  tipo_montaggio_id: number;
  stato: string;
  data_inizio: string;
  data_completamento?: string | null;
  data_fallimento?: string | null;
  responsabile_id: string;
  tentativo: number;
  note?: string | null;
  created_at: string;
  updated_at: string;
  tipi_montaggio?: { nome: string } | null;
  profiles?: { full_name: string } | null;
};

type TipoMontaggio = Database['public']['Tables']['tipi_montaggio']['Row'];

// Struttura dati per le note strutturate (JSON)
interface LabEsternoData {
  isExternalLab: boolean;
  userNotes: string;
  dates: {
    scheduled_pickup: string; // YYYY-MM-DD
    scheduled_return: string;
    actual_pickup: string | null;
    actual_return: string | null;
  }
}

interface LavorazioneTabProps {
  busta: BustaDettagliata;
  isReadOnly?: boolean;
  onBustaUpdate?: (updatedBusta: BustaDettagliata) => void;
}

// --- CONFIGURATION ---
const ACTIVITY_MATRIX: Record<string, string[]> = {
  'VISTA': ['Sagomatura & Montaggio', 'Lab. Esterno', 'Controllo Qualità Pre-Consegna', 'Richiamo per Verifica Tecnica', 'Verifica Non Adattamento', 'Gestione Assistenza/Garanzia', 'Riparazione & Minuteria', 'Pit-Stop Occhiale'],
  'SOLE': ['Controllo Qualità Pre-Consegna', 'Riparazione & Minuteria', 'Gestione Assistenza/Garanzia', 'Pit-Stop Occhiale', 'Lab. Esterno'],
  'LAC': ['Training Applicazione LAC', 'Richiamo per Verifica Tecnica', 'Gestione Assistenza/Garanzia'],
  'RIPARAZIONE': ['Riparazione & Minuteria', 'Lab. Esterno', 'Controllo Qualità Pre-Consegna']
};

const getCategoryForBusta = (type: string) => {
  if (['OCV', 'OV', 'LV'].includes(type)) return 'VISTA';
  if (['OS', 'LS', 'ACC', 'BR'].includes(type)) return 'SOLE';
  if (['LAC'].includes(type)) return 'LAC';
  if (['RIP', 'RIC', 'SA', 'SG'].includes(type)) return 'RIPARAZIONE';
  return 'VISTA';
};

const CHECKLISTS: Record<string, string[]> = {
  'Controllo Qualità Pre-Consegna': ['Pulizia lenti (no aloni)', 'Assetto aste (in piano)', 'Serraggio viti/chiusura cerchi', 'Verifica corrispondenza lenti/ordine', 'Dotazione (Astuccio + Microfibra)'],
  'Pit-Stop Occhiale': ['Lavaggio ultrasuoni', 'Sostituzione naselli (se nec.)', 'Serraggio viteria completa', 'Riadattamento assetto'],
  'Training Applicazione LAC': ['Igiene mani verificata', 'Applicazione OD riuscita', 'Applicazione OS riuscita', 'Rimozione OD riuscita', 'Rimozione OS riuscita', 'Istruzioni manutenzione fornite']
};

export default function LavorazioneTab({ busta, isReadOnly = false, onBustaUpdate }: LavorazioneTabProps) {
  const [lavorazioni, setLavorazioni] = useState<Lavorazione[]>([]);
  const [tipiMontaggio, setTipiMontaggio] = useState<TipoMontaggio[]>([]);
  const [showForm, setShowForm] = useState(false);
  const [isLoading, setIsLoading] = useState(false);
  const [isSaving, setIsSaving] = useState(false);
  const [qcCompleted, setQcCompleted] = useState(false);
  const [isMounted, setIsMounted] = useState(false);

  // Form State
  const [formData, setFormData] = useState({
    tipo_id: '',
    note: '',
    data_inizio: new Date().toISOString().split('T')[0],
    // Lab Esterno Fields
    lab_ritiro_previsto: '',
    lab_riconsegna_prevista: ''
  });
  
  const [checklistItems, setChecklistItems] = useState<string[]>([]);

  const { profile } = useUser();
  const canEdit = !isReadOnly && profile?.role !== 'operatore';
  const supabase = createBrowserClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
  );

  useEffect(() => { setIsMounted(true); }, []);
  useEffect(() => { loadData(); }, [busta.id]);
  useEffect(() => { if (busta.controllo_completato) setQcCompleted(true); }, [busta.controllo_completato]);

  const loadData = async () => {
    setIsLoading(true);
    try {
      const { data: tipi } = await supabase.from('tipi_montaggio').select('*').order('nome');
      setTipiMontaggio(tipi || []);

      const { data: works, error } = await supabase
        .from('lavorazioni')
        .select(`*, tipi_montaggio (nome), profiles (full_name)`)
        .eq('busta_id', busta.id)
        .order('created_at', { ascending: false });

      if (error) throw error;
      setLavorazioni(works as unknown as Lavorazione[]);
    } catch (err) {
      console.error(err);
    } finally {
      setIsLoading(false);
    }
  };

  const handleTypeChange = (id: string) => {
    setFormData(prev => ({ ...prev, tipo_id: id }));
    setChecklistItems([]);
  };

  const toggleChecklist = (item: string) => {
    setChecklistItems(prev => 
      prev.includes(item) ? prev.filter(i => i !== item) : [...prev, item]
    );
  };

  // --- SAVE NEW ACTIVITY ---
  const handleSave = async () => {
    if (!formData.tipo_id) return alert('Seleziona un tipo di attività');
    setIsSaving(true);

    try {
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) throw new Error("Utente non autenticato");

      const selectedType = tipiMontaggio.find(t => t.id.toString() === formData.tipo_id);
      const typeName = selectedType?.nome || '';
      
      let noteContent = formData.note.trim();

      // LOGIC: Lab Esterno (Save as JSON structure inside Note)
      if (typeName === 'Lab. Esterno') {
        const labData: LabEsternoData = {
            isExternalLab: true,
            userNotes: noteContent,
            dates: {
                scheduled_pickup: formData.lab_ritiro_previsto,
                scheduled_return: formData.lab_riconsegna_prevista,
                actual_pickup: null,
                actual_return: null
            }
        };
        // Serialize to string for DB storage
        noteContent = JSON.stringify(labData);
      }
      // LOGIC: Checklists (Append to text)
      else {
          const template = CHECKLISTS[typeName];
          if (template && checklistItems.length > 0) {
            const checklistText = template.map(item => 
              checklistItems.includes(item) ? `✅ ${item}` : `⬜ ${item}`
            ).join('\n');
            noteContent = noteContent ? `${noteContent}\n\n${checklistText}` : checklistText;
          }
      }

      const attempts = lavorazioni.filter(l => l.tipo_montaggio_id.toString() === formData.tipo_id).length + 1;
      
      const { data, error } = await supabase
        .from('lavorazioni')
        .insert({
          busta_id: busta.id,
          tipo_montaggio_id: Number(formData.tipo_id),
          stato: 'in_corso',
          data_inizio: formData.data_inizio,
          responsabile_id: user.id,
          tentativo: attempts,
          note: noteContent || null
        })
        .select(`*, tipi_montaggio(nome), profiles(full_name)`)
        .single();

      if (error) throw error;

      setLavorazioni(prev => [data as unknown as Lavorazione, ...prev]);
      setShowForm(false);
      setFormData({
        tipo_id: '', note: '', data_inizio: new Date().toISOString().split('T')[0],
        lab_ritiro_previsto: '', lab_riconsegna_prevista: ''
      });
      setChecklistItems([]);

    } catch (err: any) {
      alert(err.message);
    } finally {
      setIsSaving(false);
    }
  };

  // --- UPDATE LAB DATES (Interactive) ---
  const handleUpdateLabDates = async (workId: string, currentNoteJSON: string, field: 'actual_pickup' | 'actual_return', value: string) => {
      try {
          const data: LabEsternoData = JSON.parse(currentNoteJSON);
          data.dates[field] = value;
          
          // Logic: if actual return is set, maybe suggest completing?
          // For now, just update the data
          const newNote = JSON.stringify(data);

          const { error } = await supabase
            .from('lavorazioni')
            .update({ note: newNote })
            .eq('id', workId);

          if (error) throw error;

          // Optimistic update
          setLavorazioni(prev => prev.map(w => w.id === workId ? { ...w, note: newNote } : w));

      } catch (e) {
          console.error("Error updating lab dates", e);
          alert("Errore aggiornamento date");
      }
  };

  // --- UTILS ---
  const parseNote = (note: string | null): { isJson: boolean, content: any } => {
      if (!note) return { isJson: false, content: '' };
      try {
          if (note.trim().startsWith('{')) {
              const parsed = JSON.parse(note);
              if (parsed.isExternalLab) return { isJson: true, content: parsed };
          }
      } catch (e) { /* ignore */ }
      return { isJson: false, content: note };
  };

  const calculateDelay = (scheduled: string, actual: string | null) => {
      if (!actual) return false;
      return new Date(actual) > new Date(scheduled);
  };

  // QC Logic (Identical to previous)
  const handleQC = async (checked: boolean) => {
    if (!checked) { setQcCompleted(false); return; }
    if (!confirm("Confermi Superamento Controllo Qualità?")) return;
    setIsSaving(true);
    setQcCompleted(true);
    try {
       const { data: { user } } = await supabase.auth.getUser();
       await supabase.from('buste').update({
           controllo_completato: true,
           controllo_completato_da: user?.id,
           controllo_completato_at: new Date().toISOString()
       }).eq('id', busta.id);
       await fetch('/api/buste/update-status', {
           method: 'POST',
           body: JSON.stringify({ bustaId: busta.id, newStatus: 'pronto_ritiro', oldStatus: busta.stato_attuale })
       });
       mutate('/api/buste');
    } catch(e) { setQcCompleted(false); } finally { setIsSaving(false); }
  };

  const selectedTypeObj = tipiMontaggio.find(t => t.id.toString() === formData.tipo_id);
  const showExternalLabFields = selectedTypeObj?.nome === 'Lab. Esterno';
  const currentChecklist = selectedTypeObj ? CHECKLISTS[selectedTypeObj.nome] : null;

  return (
    <div className="space-y-6">
      
      {/* HEADER */}
      <div className="bg-white rounded-lg shadow-sm border border-gray-200 p-6">
        <div className="flex justify-between items-center">
          <div>
            <h2 className="text-xl font-bold text-gray-900 flex items-center gap-2">
              <Factory className="w-6 h-6 text-blue-600"/>
              Attività & Laboratorio
            </h2>
            <p className="text-sm text-gray-500">Gestione lavorazioni interne ed esterne.</p>
          </div>
          {isMounted && canEdit && (
            <button 
              onClick={() => setShowForm(true)}
              className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded-lg flex items-center gap-2 transition-colors"
            >
              <Plus className="w-4 h-4"/> Nuova Attività
            </button>
          )}
        </div>

        {/* --- FORM DI CREAZIONE --- */}
        {showForm && (
          <div className="mt-6 bg-gray-50 border border-gray-200 rounded-lg p-5 animate-in fade-in slide-in-from-top-2">
            <div className="flex justify-between mb-4">
              <h3 className="font-semibold text-gray-800">Registra Nuova Attività</h3>
              <button onClick={() => setShowForm(false)}><X className="w-5 h-5 text-gray-400 hover:text-gray-600"/></button>
            </div>

            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
              <div className="md:col-span-2">
                <label className="block text-xs font-bold text-gray-700 uppercase mb-1">Tipo Attività</label>
                <select 
                  className="w-full p-2.5 bg-white border border-gray-300 rounded-md focus:ring-2 focus:ring-blue-500"
                  value={formData.tipo_id}
                  onChange={e => handleTypeChange(e.target.value)}
                >
                  <option value="">-- Seleziona --</option>
                  {/* Filtro attività basato sul tipo busta (ACTIVITY_MATRIX) */}
                  {tipiMontaggio
                    .filter(t => (ACTIVITY_MATRIX[getCategoryForBusta(busta.tipo_lavorazione)] || []).includes(t.nome))
                    .sort((a, b) => {
                       const allowed = ACTIVITY_MATRIX[getCategoryForBusta(busta.tipo_lavorazione)] || [];
                       return allowed.indexOf(a.nome) - allowed.indexOf(b.nome);
                    })
                    .map(t => (
                    <option key={t.id} value={t.id}>{t.nome}</option>
                  ))}
                </select>
              </div>

              {/* === FORM SPECIALE LAB ESTERNO === */}
              {showExternalLabFields && (
                <div className="md:col-span-2 bg-blue-50 p-4 rounded-md border border-blue-100 grid grid-cols-1 md:grid-cols-2 gap-4">
                   <div className="md:col-span-2 flex items-center gap-2 text-blue-800 font-semibold text-sm mb-1">
                      <Truck className="w-4 h-4"/> Pianificazione Spedizione (Corriere Unico)
                   </div>
                   
                   <div>
                      <label className="block text-xs font-bold text-blue-900 mb-1">Data Ritiro Prevista</label>
                      <input 
                        type="date"
                        className="w-full p-2 border border-blue-200 rounded text-sm bg-white"
                        value={formData.lab_ritiro_previsto}
                        onChange={e => setFormData({...formData, lab_ritiro_previsto: e.target.value})}
                      />
                   </div>
                   <div>
                      <label className="block text-xs font-bold text-blue-900 mb-1">Data Riconsegna Prevista</label>
                      <input 
                        type="date"
                        className="w-full p-2 border border-blue-200 rounded text-sm bg-white"
                        value={formData.lab_riconsegna_prevista}
                        onChange={e => setFormData({...formData, lab_riconsegna_prevista: e.target.value})}
                      />
                   </div>
                   <p className="md:col-span-2 text-xs text-blue-600 italic">
                      * Le date effettive verranno inserite dopo la creazione, cliccando sulla scheda.
                   </p>
                </div>
              )}

              {/* CHECKLIST (Se presente per il tipo scelto) */}
              {currentChecklist && (
                <div className="md:col-span-2 bg-white p-4 rounded-md border border-gray-200">
                    <label className="block text-sm font-bold text-gray-800 mb-3 flex items-center gap-2">
                        <CheckSquare className="w-4 h-4 text-green-600"/>
                        Checklist: {selectedTypeObj?.nome}
                    </label>
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-2">
                        {currentChecklist.map((item, idx) => (
                            <label key={idx} className="flex items-center gap-2 cursor-pointer hover:bg-gray-50 p-2 rounded select-none">
                                <input 
                                    type="checkbox"
                                    checked={checklistItems.includes(item)}
                                    onChange={() => toggleChecklist(item)}
                                    className="w-4 h-4 text-blue-600 rounded border-gray-300"
                                />
                                <span className="text-sm text-gray-700">{item}</span>
                            </label>
                        ))}
                    </div>
                </div>
              )}

              <div className="md:col-span-2">
                 <label className="block text-xs font-medium text-gray-700 mb-1">Note / Dettagli</label>
                 <textarea 
                    className="w-full p-2.5 border border-gray-300 rounded-md"
                    rows={2}
                    placeholder="Note libere..."
                    value={formData.note}
                    onChange={e => setFormData({...formData, note: e.target.value})}
                 />
              </div>
            </div>

            <div className="mt-5 flex justify-end gap-3 pt-4 border-t border-gray-200">
              <button onClick={() => setShowForm(false)} className="px-4 py-2 text-gray-600 hover:bg-gray-100 rounded-lg">Annulla</button>
              <button 
                onClick={handleSave}
                disabled={isSaving}
                className="px-4 py-2 bg-blue-600 hover:bg-blue-700 text-white rounded-lg flex items-center gap-2"
              >
                {isSaving && <Loader2 className="w-4 h-4 animate-spin"/>}
                Salva Attività
              </button>
            </div>
          </div>
        )}
      </div>

      {/* --- LISTA ATTIVITÀ --- */}
      <div className="space-y-4">
        {lavorazioni.map(work => {
            const parsedNote = parseNote(work.note);
            
            return (
              <div key={work.id} className="bg-white rounded-lg shadow-sm border border-gray-200 overflow-hidden">
                {/* Header Card */}
                <div className="bg-gray-50 p-3 border-b border-gray-200 flex justify-between items-center">
                    <div className="flex items-center gap-2">
                        <h4 className="font-bold text-gray-900 text-sm">{work.tipi_montaggio?.nome}</h4>
                        <span className={`text-[10px] uppercase font-bold px-2 py-0.5 rounded ${
                            work.stato === 'completato' ? 'bg-green-100 text-green-700' : 
                            work.stato === 'fallito' ? 'bg-red-100 text-red-700' : 'bg-blue-100 text-blue-700'
                        }`}>{work.stato.replace('_', ' ')}</span>
                    </div>
                    {canEdit && (
                         <button 
                            onClick={async () => {
                              if(confirm('Eliminare questa voce?')) {
                                await supabase.from('lavorazioni').delete().eq('id', work.id);
                                setLavorazioni(prev => prev.filter(l => l.id !== work.id));
                              }
                            }}
                            className="text-gray-400 hover:text-red-500"
                         ><Trash2 className="w-4 h-4"/></button>
                    )}
                </div>

                <div className="p-4">
                    {/* --- RENDERING SPECIALE PER LAB ESTERNO --- */}
                    {parsedNote.isJson ? (
                        <div className="space-y-4">
                            {/* Griglia Date */}
                            <div className="grid grid-cols-1 md:grid-cols-2 gap-6 bg-blue-50/50 p-4 rounded-lg border border-blue-100">
                                
                                {/* Colonna 1: Ritiri */}
                                <div className="space-y-3">
                                    <div className="flex items-center gap-2 text-blue-900 font-semibold text-sm border-b border-blue-200 pb-1">
                                        <ArrowRight className="w-4 h-4"/> Spedizione (Ritiro)
                                    </div>
                                    <div className="grid grid-cols-2 gap-2 text-xs">
                                        <div>
                                            <span className="block text-gray-500">Prevista:</span>
                                            <span className="font-medium text-gray-900">
                                                {parsedNote.content.dates.scheduled_pickup ? new Date(parsedNote.content.dates.scheduled_pickup).toLocaleDateString() : '-'}
                                            </span>
                                        </div>
                                        <div>
                                            <span className="block text-gray-500">Effettiva:</span>
                                            {canEdit ? (
                                                <input 
                                                    type="date"
                                                    className={`w-full p-1 border rounded bg-white text-xs ${
                                                        calculateDelay(parsedNote.content.dates.scheduled_pickup, parsedNote.content.dates.actual_pickup)
                                                        ? 'border-red-300 text-red-600 font-bold'
                                                        : 'border-gray-300 text-gray-900'
                                                    }`}
                                                    value={parsedNote.content.dates.actual_pickup || ''}
                                                    onChange={(e) => handleUpdateLabDates(work.id, work.note!, 'actual_pickup', e.target.value)}
                                                />
                                            ) : (
                                                <span className={calculateDelay(parsedNote.content.dates.scheduled_pickup, parsedNote.content.dates.actual_pickup) ? 'text-red-600 font-bold' : ''}>
                                                    {parsedNote.content.dates.actual_pickup ? new Date(parsedNote.content.dates.actual_pickup).toLocaleDateString() : '-'}
                                                </span>
                                            )}
                                        </div>
                                    </div>
                                </div>

                                {/* Colonna 2: Riconsegne */}
                                <div className="space-y-3">
                                    <div className="flex items-center gap-2 text-blue-900 font-semibold text-sm border-b border-blue-200 pb-1">
                                        <Truck className="w-4 h-4"/> Rientro in Sede
                                    </div>
                                    <div className="grid grid-cols-2 gap-2 text-xs">
                                        <div>
                                            <span className="block text-gray-500">Prevista:</span>
                                            <span className="font-medium text-gray-900">
                                                {parsedNote.content.dates.scheduled_return ? new Date(parsedNote.content.dates.scheduled_return).toLocaleDateString() : '-'}
                                            </span>
                                        </div>
                                        <div>
                                            <span className="block text-gray-500">Effettiva:</span>
                                            {canEdit ? (
                                                <input 
                                                    type="date"
                                                    className={`w-full p-1 border rounded bg-white text-xs ${
                                                        calculateDelay(parsedNote.content.dates.scheduled_return, parsedNote.content.dates.actual_return)
                                                        ? 'border-red-300 text-red-600 font-bold'
                                                        : 'border-gray-300 text-gray-900'
                                                    }`}
                                                    value={parsedNote.content.dates.actual_return || ''}
                                                    onChange={(e) => handleUpdateLabDates(work.id, work.note!, 'actual_return', e.target.value)}
                                                />
                                            ) : (
                                                <span className={calculateDelay(parsedNote.content.dates.scheduled_return, parsedNote.content.dates.actual_return) ? 'text-red-600 font-bold' : ''}>
                                                    {parsedNote.content.dates.actual_return ? new Date(parsedNote.content.dates.actual_return).toLocaleDateString() : '-'}
                                                </span>
                                            )}
                                        </div>
                                    </div>
                                </div>
                            </div>
                            
                            {/* Alert Ritardo */}
                            {(calculateDelay(parsedNote.content.dates.scheduled_return, parsedNote.content.dates.actual_return) || 
                              calculateDelay(parsedNote.content.dates.scheduled_pickup, parsedNote.content.dates.actual_pickup)) && (
                                <div className="flex items-center gap-2 text-xs text-red-600 bg-red-50 p-2 rounded">
                                    <AlertCircle className="w-4 h-4"/> Attenzione: Ritardo registrato rispetto alla pianificazione.
                                </div>
                            )}

                            {/* Note Aggiuntive JSON */}
                            {parsedNote.content.userNotes && (
                                <div className="text-sm text-gray-600 italic border-t pt-2 mt-2">
                                    Note: {parsedNote.content.userNotes}
                                </div>
                            )}
                        </div>
                    ) : (
                        // --- RENDERING STANDARD (Testo / Checklist) ---
                        work.note && (
                            <div className="text-sm text-gray-700 whitespace-pre-wrap font-mono">
                                {work.note}
                            </div>
                        )
                    )}
                    
                    <div className="text-xs text-gray-400 mt-4 flex justify-between items-center">
                        <span>Resp: {work.profiles?.full_name}</span>
                        <span>Creato: {new Date(work.created_at).toLocaleDateString()}</span>
                    </div>
                </div>
              </div>
            );
        })}
      </div>

      {/* QC CHECKBOX */}
      {showForm === false && (
          <div className={`p-4 rounded-lg border-2 transition-colors mt-6 ${
              qcCompleted 
              ? 'bg-emerald-50 border-emerald-400' 
              : 'bg-white border-gray-200'
          }`}>
              <div className="flex items-start gap-3">
                  <input 
                    type="checkbox"
                    id="qc_check"
                    checked={qcCompleted}
                    onChange={e => handleQC(e.target.checked)}
                    disabled={busta.controllo_completato}
                    className="mt-1 w-5 h-5 text-emerald-600 rounded border-gray-300 focus:ring-emerald-500 cursor-pointer"
                  />
                  <div>
                      <label htmlFor="qc_check" className="font-bold text-gray-900 cursor-pointer select-none text-lg">
                          Certifico Controllo Qualità Finale
                      </label>
                      <p className="text-sm text-gray-600 mt-1">
                          {qcCompleted 
                            ? `Certificato da ${busta.controllo_profile?.full_name} il ${new Date(busta.controllo_completato_at!).toLocaleDateString()}.`
                            : "Confermo che il prodotto è conforme, pulito e completo di dotazione."
                          }
                      </p>
                  </div>
              </div>
          </div>
      )}

    </div>
  );
